        Prints the list elements sorted in ascending order.

        Doctests:
        >>> my_list = MyList([3, 2, 1, 5, 4])
        >>> my_list.print_sorted()
        [1, 2, 3, 4, 5]
        >>> my_list = MyList(['c', 'a', 'b'])
        >>> my_list.print_sorted()
        ['a', 'b', 'c']
        >>> my_list = MyList([])
        >>> my_list.print_sorted()
        []
        >>> my_list = MyList([5, 3, 4, 1, 2])
        >>> my_list.print_sorted()
        [1, 2, 3, 4, 5]
        >>> my_list = MyList(['e', 'c', 'a', 'd', 'b'])
        >>> my_list.print_sorted()
        ['a', 'b', 'c', 'd', 'e']
